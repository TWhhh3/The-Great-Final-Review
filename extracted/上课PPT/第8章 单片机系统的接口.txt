     当按键在一次被按下的过程中，其功能程序被反复多次
执行的现象，好像按键被多次按下一样，这种现象就称为

         键盘中每个按键都是一个常开开关电路，是利用机械触点来实现按键的闭合和释放。

     由于弹性作用的影响，按键的

电压信号也出现抖动现象。

各键是相互独立的。当某个键按

1）快速扫描判别是否有键按下。通过行线送出扫描字0000B，然后读入列线状态，假如读入的
列线端口值全为1，则说明没有按键被按下，反之则说明有键按下。
     2）调用延时（或者是执行其他任务来用作延时）去除抖动。当检测到有键按下后，软件延时一
段时间，然后再次检测按键的状态，这时检测到仍有按键被按下，则就可以认为按键确实被按下了，
否则只能按照按键抖动来处理。
     3）按键的键值处理。当有键按下时，就可进入逐行扫描的方法来确定到底是哪一个按键被按下。
先扫描第一行，即将第一行输出为低电平（0），然后再去读入列线的端口值，如果哪一列出现低
电平（0），就说明该列与第一行跨接的按键被按下了。如果读入的列线端口值全为1，则说明与第
一行跨接的所有按键都没有被按下。接着就扫描第二行，以此类推，逐行扫描，直至找到被按下的
按键，并根据事先的定义将按键的键值送入键值变量中保存。需要注意的是，在返回键盘的键值前
还需要检测按键是否释放，这样可以避免连击现象的出现，保证每次按键只做一次处理。
       4）返回按键的键值的处理。根据按键的编码值，就可以进行相应按键的功能处理（本例仅显
示对应按键按下时的序号，实际应用中需要设计该键执行的功能程序）。

//按键函数：整体扫描判定是否有键按下（返回值不等于0x0，说明有键按下）如果为0表示无键按下。
// 键盘矩阵（P2.0~2.3）全置零，读入后(整体)判断是否有键按下,
uchar keysearch()
{  uchar k;
 P2=0xf0;  // P2.4~2.7 是按键矩阵的列，待读入；P2.0~2.3是行，读入前置零
 k=P2;
 k=~k;    //按键按下时返回0，取反，如果键被按下，则键值 为1
 k=k & 0xf0;   // P2.4~2.7 高4位是返回值，屏蔽低4位P2.0~2.3
 return k;
}

uchar key()   //返回按键的编码键值,对键盘矩阵逐行扫描，判定具体是哪个键被按下
{ uchar a,c,kr,keynumb;
 a=keysearch();  //按键整体扫描函数，第一次读入
 if(a==0)    return 0xff;     // 无键按下
 else  delay(10);  //有，延时去抖动
 a=keysearch();     //按键整体扫描，第2次读入
 if(a==0)    return 0xff;  // 无键按下
 else    // 有键按下
{  a=0xfe;  // 只令一行为0(低电平)，通过循环左移，对键盘矩阵逐行扫描 
for(kr=0;kr<4;kr++)   // 4行: 0~3
{ P2 = a;   // 只令其中一行为0(低电平)
c = P2;   // 读入
if((c & 0x10)==0) keynumb=kr+0x00;  // 第1列？按键编码：行号(kr)+0x00
if((c & 0x20)==0) keynumb=kr+0x04;  // 第2列？按键编码：行号+0x04
if((c & 0x40)==0) keynumb=kr+0x08;  // 第3列？按键编码：行号+0x08
if((c & 0x80)==0) keynumb=kr+0x0c;  // 第4列？按键编码：行号+0x0c
a=_crol_(a,1); //循环左移函数,需要intrins.h头文件支持，逐行扫描
}
}
do{ a=keysearch();  }while(a!=0);   //等待按键松开，未松开则等待
return keynumb;  //返回按键的编码键值
}

#include <reg52.h>   //头文件
#define uchar unsigned char //宏定义
#define uint unsigned int
sbit lcden=P3^1;    // E端口定义，由高跳变成低电平时，液晶模块执行命令  
sbit lcdrs=P3^0;    //  RS数据/命令选择,  高:选择数据寄存器, 低:选择指令寄存器
//*******************************************************************************
//延时函数    
void delay(uint x)
{
uint i,j;
for(i=0;i<x;i++)
       for(j=0;j<120;j++);
}

/*******************************************************************************
 写控制指令
*******************************************************************************/
void write_com(uchar com)
{
lcdrs=0;    // lcdrs为低电平时,写控制命令
delay(1);    //稍作延时以待数据稳定
P1=com;           // 从P1输出(送 LCD1602的D0~D7)
lcden=1;             // E 使能
delay(1);
lcden=0;             //  由高跳变成低电平时，执行命令；E低电平时 禁止
}

/*******************************************************************************
写待显示的数据
*******************************************************************************/
void write_data(uchar dat)
{
lcdrs=1;  //lcdrs为高电平时,写要显示的数据
delay(1); //稍作延时以待数据稳定
P1=dat;         // 从P1输出(送 LCD1602的D0~D7)
lcden=1;         // E 使能       
delay(1);
lcden=0;          //  由高跳变成低电平时，执行命令；E低电平时 禁止
}

/******************************************************************************
写连续字符函数
******************************************************************************/
void write_word(uchar *s)
{
while(*s>0)
{
       write_data(*s);
       s++;
}
}

/*******************************************************************************
初始化
*******************************************************************************/
void init()
{
lcden=0;
write_com(0x38); //显示模式设定: 8为数据线，2行(x16)，5x7点阵字符,---指令6
write_com(0x0c); //开关显示、光标有无设置、光标闪烁设置  ，--指令4
write_com(0x06); //写一个字符后指针加一 ，--指令3
write_com(0x01); //清屏指令 ，--指令1
}
